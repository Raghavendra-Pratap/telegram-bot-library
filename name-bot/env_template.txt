# Telegram Bot Configuration
# Copy this file to .env and fill in your values

# Required: Your Telegram Bot Token
# Get it from @BotFather on Telegram
TELEGRAM_BOT_TOKEN=your_bot_token_here

# Optional: User Access Control
# Set to true to restrict bot usage to specific users
ENABLE_USER_VERIFICATION=false

# Optional: Allowed User IDs (comma-separated)
# Only used if ENABLE_USER_VERIFICATION=true
# Get your user ID from @userinfobot on Telegram
ALLOWED_USER_IDS=

# Optional: Skip adding caption when original filename is not available
# If true, bot won't add caption for mobile uploads where Telegram doesn't provide filename
# If false, bot will use generated filename (file_id based) as caption
SKIP_IF_NO_FILENAME=false

# Optional: Retry configuration (optimized for large batches)
# Delay between retries when errors occur (in seconds)
RETRY_DELAY=2.0

# Optional: Maximum retry attempts (increased for flood control)
# Higher value = more attempts before giving up (useful for rate limits)
MAX_RETRIES=5

# Optional: Flood control retry delay multiplier
# When flood control is hit, wait this multiplier × requested time
# 1.5 = wait 50% longer than Telegram requests (safer)
FLOOD_RETRY_DELAY_MULTIPLIER=1.5

# Optional: Parallel Processing Configuration
# Maximum number of files to process simultaneously
# Telegram allows ~20-30 messages/second, so 10-15 concurrent tasks is safe
# Higher values = faster processing but higher risk of rate limits
# Recommended: 10 for normal use, 15-20 for high-traffic scenarios
MAX_CONCURRENT_TASKS=10

# Optional: Enable concurrent updates for parallel processing
# Set to true to process multiple files simultaneously (recommended)
# Set to false to process files one at a time (sequential)
ENABLE_CONCURRENT_UPDATES=true

# Optional: Minimum delay between API calls within the same task (in seconds)
# This prevents too many rapid API calls from a single task
# Lower values = faster but may hit rate limits, higher = safer but slower
# Recommended: 0.1 (100ms) for parallel processing, 0.5-1.0 for sequential
MIN_API_CALL_DELAY=0.1

# Optional: Auto-Shutdown Configuration (for resource-saving local setups)
# Idle timeout in minutes - bot will shut down after this period of inactivity
# Set to 0 to disable auto-shutdown (bot runs continuously)
# Recommended: 0 for always-on servers, or 10-15 minutes to save resources
IDLE_TIMEOUT_MINUTES=12

# Optional: Enable auto-shutdown feature
# Set to true to enable idle timeout and auto-shutdown (saves resources)
# Set to false to run bot continuously (traditional mode)
ENABLE_AUTO_SHUTDOWN=true

# Optional: HTTP Server Configuration (local health checks)
# Port for HTTP health check server (useful for LAN monitoring)
# Default: 8080
HTTP_SERVER_PORT=8080

# Optional: Enable HTTP server for health checks and wake-up
# Set to true to enable HTTP server (recommended for local monitoring)
# Set to false to disable HTTP server
ENABLE_HTTP_SERVER=true

# Optional: Update Queue Configuration
# Enable processing of missed updates when bot restarts
# When enabled, bot will fetch and process updates that arrived while bot was offline
# This ensures no files are missed when bot was shut down
ENABLE_UPDATE_QUEUE=true

# Optional: Maximum number of updates to process from queue on startup
# Telegram stores updates for up to 24 hours
# Higher values = process more missed updates, but slower startup
# Recommended: 100 for normal use, 50 for faster startup
MAX_QUEUE_UPDATES=100
